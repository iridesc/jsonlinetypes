def test_batch_operations():
    """测试批量操作的性能和正确性"""
    d = JLFDict("batch.jsonl", "id")

    # 方法 1：逐个操作
    start = time.time()
    for i in range(1000):
        d[f"key_{i}"] = {"id": i, "value": i}
    time_individual = time.time() - start

    # 方法 2：批量操作
    d.clear()
    start = time.time()
    with d:
        for i in range(1000):
            d[f"key_{i}"] = {"id": i, "value": i}
    time_batch = time.time() - start

    # 批量操作应该更快（减少锁）
    # 性能提升取决于系统
    print(f"逐个操作: {time_individual:.3f}s")
    print(f"批量操作: {time_batch:.3f}s")
